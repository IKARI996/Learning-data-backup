#include<iostream>
#include<queue>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        queue<int> q;
        for(int i=1;i<=n;++i){
            q.push(i);
        }
        while(q.size()){
            cout<<q.front()<<" ";
            q.pop();
            int temp = q.front();
            q.pop();
            q.push(temp);
        }
        cout << endl;
    }
}
