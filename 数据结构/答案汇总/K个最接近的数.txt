#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> findClosestElements(vector<int>& arr, int k, int x) {
	auto h = arr.size() - k;
	auto l = 0;
	while (l < h) {
		auto mid = (l + h) / 2;
		if (abs(arr[mid] - x) <= abs(arr[k + mid] - x)) {
			h = mid;
		}
		else {
			l = mid + 1;
		}
	}
	vector<int> rst(arr.begin() + h, arr.begin() + h + k);
	return rst;
}
int main()
{
	int n;
	cin >> n;
	vector<int> v;
	for (int i = 0; i < n; ++i)
	{
		int temp;
		cin >> temp;
		v.push_back(temp);
	}
	int k, x;
	cin >> k >> x;
	vector<int> result = findClosestElements(v, k, x);
		cout << result[0];
	for (int i = 1; i < result.size(); ++i)
	{
		cout << ' ' << result[i];
	}
	return 0;
}
