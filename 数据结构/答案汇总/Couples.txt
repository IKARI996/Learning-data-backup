#include <iostream>
#include <stack>
#include <vector>
using namespace std;

int main()
{
    int t,n,x,y;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int q=0;
        vector<int> v(2*n+1,-1);
        for(int i=0;i<n;++i)
        {
            cin>>x>>y;
            v[x]=v[y]=q++;
        }
        
        stack<int> s;
        for(int i=1;i<=2*n;++i)
        {
            if(!s.empty()&&v[i]==v[s.top()]) s.pop();
            else s.push(i);
        }
        if(s.empty()) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }           
    return 0;
}
