#include <iostream>
#include <string>
using namespace std;
int main()
{
	string sstr;
	string tstr;
	cin >> sstr >> tstr;
	int count = 0;
	int index = tstr.find(sstr);
	while (index != -1)
	{
		count++;
		if (tstr[index + 1] == '\0') break;
		index = tstr.find(sstr, index + 1);
	}
	cout << count;
	return 0;
}
