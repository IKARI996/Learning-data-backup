;THIS PROGRAM FINDS THE SUM OF TWO EXTERNALLY DEFINED WORDS
;AND STORES THE SUM IN A LOCATION DEFINED BY THE CALLING MODULE
TITLE    SUBPROG1 PROGRAM TO ADD TWO WORDS 
PAGE     60,132
         EXTRN VALUE1:WORD
         EXTRN VALUE2:WORD
         EXTRN SUM:WORD
         PUBLIC SUBPROG1
CODSG    SEGMENT PARA PUBLIC 'CODE'
SUBPROG1 PROC    FAR
         ASSUME CS:CODSG
         SUB     BX,BX          ;INITIALIZE CARRY COUNT
         MOV     AX,VALUE1
         MOV     DX,VALUE2
         ADD     AX,DX          ;ADD VALUE1 + VALUE2
         ADC     BX,00          ;ACCUMULATE CARRY
         MOV     SUM,AX         ;STORE SUM
         MOV     SUM+2,BX       ;STORE CARRY
         RET
SUBPROG1 ENDP
CODSG    ENDS
         END
